function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss   = SpreadsheetApp.getActiveSpreadsheet();
    var key  = data.key   || 'DASHBOARD_SYNC';
    var val  = data.value || '';
    var tab  = ss.getSheetByName('__DASHBOARD_SYNC__')
               || ss.insertSheet('__DASHBOARD_SYNC__');
    // Find or create row for this key
    var found = false;
    var lastRow = tab.getLastRow();
    for (var i = 1; i <= lastRow; i++) {
      if (tab.getRange(i,1).getValue() === key) {
        tab.getRange(i,2).setValue(val);
        tab.getRange(i,3).setValue(new Date().toISOString());
        found = true; break;
      }
    }
    if (!found) {
      tab.appendRow([key, val, new Date().toISOString()]);
    }
    return ContentService
      .createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(ex) {
    return ContentService
      .createTextOutput(JSON.stringify({ok:false,error:ex.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
